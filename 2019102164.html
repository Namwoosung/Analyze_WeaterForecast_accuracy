import csv
#csv  csv import

location = ['maetan1','maetan2','maetan3','maetan4','woncheon','yeongtong1','yeongtong2','taejang','gwanggyo1','gwanggyo2']
year = ['2016','2017','2018']
rain = {}
#    10 2016 2018      location year 
#      rain   
for i in location:
    for j in year:
        data_lst=[]
        information = open('{0}_{1}.csv'.format(i,j),'r')
        lines = csv.reader(information)
        #   csv  
        #csv          data_lst   
        k = 0
        # k       
        for line in lines:
        #csv    
            del line[1]
            #          
            if line[1] == '0' or line[0] == ' ':
                continue
            #    (0)    ,   
            else :
                data_lst.append(line)
            #     data_lst 
            if 'Start' in line[1] or 'Start' in line[0]:
                k = k + 1
                data_lst.remove(line)
            else:
                line[0] = str(int(('{0}'.format(j))+str(k).zfill(2)+'01')+int(line[0])-1)
            #csv      ..  
            for test in data_lst:
                num_test = data_lst.count(test)
                if num_test>1:
                    for a in range(num_test-1):
                        data_lst.remove(test)
            data_lst.sort()
            #   ,     ,   
            rain[i+'_'+j]=data_lst
            #  rain  
        information.close()
        #csv
